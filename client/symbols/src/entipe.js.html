1 "use strict";
  2 
  3 const CLEAN = 0;
  4 const DIRTY = 1;
  5 
  6 /**
  7  * Schema
  8  *
  9  * @class
 10  *
 11  * A Schema object is virtually a database handle.  It knows all
 12  * entities and attributes defined in the associated database and can
 13  * be used to create a data access object for each entity.
 14  *
 15  * @param {string} url The URL of the Entipe server.
 16  * @param {Object} schema The database schema.
 17  */
 18 var Schema = function (url, schema)
 19 {
 20   // Make the schema object available for all child objects.
 21 
 22   let this_schema = this;
 23 
 24   // Quote an SQL string.
 25 
 26   let quote_string = function (string) {
 27     string = String(string);
 28     return "'" + string.replace(/'/g, "''") + "'";
 29   };
 30 
 31   // Verify an SQL identifier.  This function makes sure that SQL
 32   // identifiers, which would cause a colision with JavaScript
 33   // identifiers, are rejected.
 34 
 35   let verify_identifier = function (identifier) {
 36     identifier = String(identifier);
 37     if (identifier.match(/"$/) || identifier === '__proto__') {
 38       throw new Error("Unsupported identifier", identifier);
 39     }
 40     return identifier;
 41   };
 42 
 43   // Quote an SQL identifier.
 44 
 45   let quote_identifier = function (identifier) {
 46     return '"' + identifier + '"';
 47   };
 48 
 49   // Execute an AJAX query to the Entipe server.
 50 
 51   let query = function (query, success, error) {
 52     return jQuery.ajax({
 53       'type': 'POST',
 54       'url': url,
 55       encoding: 'UTF-8',
 56       'contentType': 'application/sql; charset=UTF-8',
 57       'data': query,
 58       'dataType': 'json',
 59       'success': success,
 60       'error': error});
 61   };
 62 
 63   /**
 64    * Execute a select query to the Entipe server and select entities
 65    * of the specified type with the given condition.
 66    */
 67   let select = function (entity, condition, success, error) {
 68     if (schema[entity]) {
 69       let q = "select * from " + quote_identifier(entity);
 70       if (condition)  q += " where " + condition;
 71       console.debug (q);
 72       query (q,
 73              success,
 74              error);
 75     } else {
 76       throw new Error ("Unknown entity " + entity);
 77     }
 78   };
 79 
 80   // '$' is the name of the object, which contains for each entity a
 81   // search function, which selects existing entities from the
 82   // database.
 83 
 84   this.$ = {};
 85 
 86   for (let entity in schema)
 87   {
 88     verify_identifier(entity);
 89 
 90     let attributes = schema[entity].map(verify_identifier);
 91 
 92     this.$[entity] = function (condition, success, error) {
 93       select (entity, condition, success, error);
 94     };
 95 
 96     /**
 97      * Entity
 98      *
 99      * @class
100      *
101      * Creates a data access object for a database entity.  The
102      * attributes of the object passed to the constructor must match
103      * the attributes defined during the schema definition.
104      *
105      * @param {Object} values The attribute values.
106      */
107     let Entity = function (values)
108     {
109       let this_entity = this;
110       let id;
111       let state = {};
112 
113       let insert = function (undo)
114       {
115         let d = []; // dirty attributes
116         let i = []; // identfier names of dirty attributes
117         let v = []; // values of dirty attributes
118         for (let a in this_entity) {
119           if (state[a] == DIRTY) {
120             d.push (a);
121             i.push(quote_identifier(a));
122             v.push(quote_string(values[a]));
123           }
124         }
125         let q
126             = "insert into " + quote_identifier(entity)
127             + " (" + i.join() + ")"
128             + " values (" + v.join() + ")";
129         console.debug ("Insert: ", q);
130         query (
131           q,
132           function(data) {
133             id = data[0].id;
134             for (let a in d)
135               state[a] = CLEAN;
136           },
137           function() {
138             let undo_error;
139             try {
140               if (undo) undo();
141             }
142             catch (e) {
143               undo_error = e;
144               console.error ("undo failed", e);
145             }
146             finally {
147               if (undo_error)
148                 throw new Error ("Insert and undo failed", undo_error);
149               else
150                 throw new Error ("Insert failed");
151             }
152           });
153       };
154 
155       Object.defineProperty (this_entity, "id", {
156         get: function () { return id; }});
157 
158       for (let attribute of attributes) {
159         state[attribute] = DIRTY;
160         Object.defineProperty (this_entity, attribute, {
161           enumerable: true,
162           get: function () {
163             return values[attribute];
164           },
165           set: function (value) {
166             let undo;
167             {
168               let old = values[attribute];
169               undo = function() { values[attribute] = old; };
170             }
171             values[attribute] = value;
172             if (id) {
173               let q
174                   = "update " + quote_identifier(entity)
175                   + " set " + quote_identifier(attribute)
176                   + " = " + quote_string(value)
177                   + " where id = " + this_entity.id;
178               console.debug ("Update: ", q);
179               let r = query (q, function (data) {}, undo);
180               console.debug ("Result: ", r);
181             }
182             else
183               insert(undo);
184             return;
185           }});
186       }
187 
188       insert ();
189     };
190 
191     this[entity] = Entity;
192   }
193 };
